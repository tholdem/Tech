\begin{defn}
$ \text{Diff}(M) $ is a topological group under smooth convergence on compact sets.
\end{defn}
Assume $ M$ is compact.

 \begin{prop}
$ \text{Diff}(M) $ is locally homeomorphic to $ \ell^2$, \emph{i.e.} separable Hilbert space. In particular, it is locally contractible.
\end{prop}
